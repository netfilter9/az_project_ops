REPORT zcm_smt2_exp_trusted_rfc.

TYPES: BEGIN OF id_server_struc,
         rfcsysid     TYPE rfcssysid,
         tlicense_nr  TYPE slic_inst,
         rfcslopt     TYPE rfcslopt,
         name         TYPE msname2,
         host         TYPE mshost2,
         serv         TYPE msserv,
         msgtypes     TYPE mstypes,
         hostadr      TYPE mshostadr,
         servno       TYPE msservno,
         state        TYPE msstate,
         no_authority TYPE abap_bool,
       END OF id_server_struc.

CONSTANTS: lc_sep          TYPE c VALUE '|'             LENGTH 1,
           lc_null         TYPE c VALUE 'NULL'          LENGTH 4,
           lc_export       TYPE c VALUE 'EXPORT'        LENGTH 6,
           lc_rfctrustid   TYPE c VALUE 'RFCTRUSTID'    LENGTH 14,
           lc_tlicense_nr  TYPE c VALUE 'TLICENSE_NR'   LENGTH 12,
           lc_rfctrustsy   TYPE c VALUE 'RFCTRUSTSY'    LENGTH 14,
           lc_llicense_nr  TYPE c VALUE 'LLICENSE_NR'   LENGTH 14,
           lc_rfcdest      TYPE c VALUE 'RFCDEST'       LENGTH 7,
           lc_rfccredest   TYPE c VALUE 'RFCCREDEST'    LENGTH 14,
           lc_rfcregdest   TYPE c VALUE 'RFCREGDEST'    LENGTH 14,
           lc_rfcmsgsrv    TYPE c VALUE 'RFCMSGSRV'     LENGTH 12,
           lc_called_sys   TYPE c VALUE 'CalledSystems' LENGTH 14,
           lc_status       TYPE c VALUE 'Status'        LENGTH 6,
           lc_description  TYPE c VALUE 'Description'   LENGTH 15,
           lc_status_check TYPE c VALUE 'StatusCheck'   LENGTH 15,
           lc_install_no   TYPE c VALUE
           'Installationnumber'                         LENGTH 20,
           lc_title        TYPE c VALUE
           'Trusted Trusting Connections'               LENGTH 35,
           lc_err_msg      TYPE c VALUE
           'No Trusting RFC found in the system'        LENGTH 40
           ##NO_TEXT.

DATA: ls_input    TYPE tab512,
      ls_output   TYPE tab512,
      lv_action   TYPE char20,
      lv_state    TYPE char2,
      lt_rfctrust TYPE TABLE OF rfctrust,
      ls_rfctrust TYPE rfctrust,
      lt_server   TYPE TABLE OF id_server_struc,
      ls_server   TYPE id_server_struc,
      lt_trusting TYPE cl_rfc_tt_factory=>rfc_sys_license_tab,
      ls_trusting TYPE LINE OF cl_rfc_tt_factory=>rfc_sys_license_tab.

LOAD-OF-PROGRAM.

FORM get_config
              TABLES lt_input  STRUCTURE tab512
                     lt_output STRUCTURE tab512.

  READ TABLE lt_input INTO ls_input INDEX 1.
  CHECK sy-subrc EQ 0.
  lv_action = ls_input.      " Action

  CASE lv_action.

    WHEN lc_export.               " When action is EXPORT

      SELECT rfctrustid tlicense_nr rfctrustsy
             llicense_nr rfcdest rfccredest
             rfcmsgsrv rfcregdest
             FROM rfctrust
             INTO CORRESPONDING FIELDS OF
             TABLE lt_rfctrust.

       " Adding title to the output
        ls_output = lc_title.
        APPEND ls_output TO lt_output.

      IF lt_rfctrust IS NOT INITIAL.

        " Sorting table by RFCSYSSID
        SORT lt_rfctrust BY rfctrustid.

        " Adding heading to the output
        CONCATENATE lc_rfctrustid   lc_sep
                    lc_tlicense_nr  lc_sep
                    lc_rfctrustsy   lc_sep
                    lc_llicense_nr  lc_sep
                    lc_rfcdest      lc_sep
                    lc_rfccredest   lc_sep
                    lc_rfcregdest   lc_sep
                    lc_rfcmsgsrv
                    INTO ls_output.
        APPEND ls_output TO lt_output.

        LOOP AT lt_rfctrust INTO ls_rfctrust.

          CONCATENATE ls_rfctrust-rfctrustid   lc_sep
                      ls_rfctrust-tlicense_nr  lc_sep
                      ls_rfctrust-rfctrustsy   lc_sep
                      ls_rfctrust-llicense_nr  lc_sep
                      ls_rfctrust-rfcdest      lc_sep
                      ls_rfctrust-rfccredest   lc_sep
                      ls_rfctrust-rfcregdest   lc_sep
                      ls_rfctrust-rfcmsgsrv
                      INTO ls_output.
          APPEND ls_output TO lt_output.

        ENDLOOP.

      ELSE.

        " No Trusting RFC found in the system
        ls_output = lc_err_msg.
        APPEND ls_output TO lt_output.

      ENDIF.

      ls_output = space.
      APPEND ls_output TO lt_output.


      TRY.
         " Method to fetch Trusting Rfc list
          CALL METHOD cl_rfc_tt_factory=>get_trusting_list
            CHANGING
              trusting_list = lt_trusting.
        CATCH cx_rfc_trusted_trusting .
      ENDTRY.


      LOOP AT lt_trusting INTO ls_trusting.
        ls_server-rfcsysid  = ls_trusting-rfcsysid.
        ls_server-tlicense_nr = ls_trusting-tlicense_nr.
        APPEND ls_server TO lt_server.
      ENDLOOP.


      TRY.
         " Method to perform Status Check
          CALL METHOD cl_rfc_tt_factory=>check_trusting_on_trusted
            CHANGING
              id_server_list = lt_server.

        CATCH cx_rfc_trusted_trusting.

      ENDTRY.


      IF lt_server IS NOT INITIAL.

        " Constructing title and heading to the output
        ls_output = lc_status_check.
        APPEND ls_output TO lt_output.

        CONCATENATE lc_called_sys   lc_sep
                    lc_install_no   lc_sep
                    lc_status       lc_sep
                    lc_description
                    INTO ls_output.
        APPEND ls_output TO lt_output.


        LOOP AT lt_server INTO ls_server.

          lv_state = ls_server-state.

          CONCATENATE ls_server-rfcsysid      lc_sep
                      ls_server-tlicense_nr   lc_sep
                      lv_state                lc_sep
                      ls_server-rfcslopt
                      INTO ls_output.
          APPEND ls_output TO lt_output.

        ENDLOOP.

      ENDIF.

  ENDCASE.

ENDFORM.
