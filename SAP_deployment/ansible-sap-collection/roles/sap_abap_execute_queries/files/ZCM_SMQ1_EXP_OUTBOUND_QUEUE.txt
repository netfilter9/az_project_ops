REPORT zcm_smq1_exp_outbound_queue.

CONSTANTS: lc_sep         TYPE c VALUE '|'              LENGTH 1,
           lc_null        TYPE c VALUE 'NULL'           LENGTH 4,
           lc_export      TYPE c VALUE 'EXPORT'         LENGTH 6,
           lc_client      TYPE c VALUE 'Client'         LENGTH 6,
           lc_queue_name  TYPE c VALUE 'QueueName'      LENGTH 12,
           lc_destination TYPE c VALUE 'Destination'    LENGTH 25,
           lc_entries     TYPE c VALUE 'Entries'        LENGTH 10,
           lc_title       TYPE c VALUE 'Outbound Queue' LENGTH 25,
           lc_err_msg     TYPE c VALUE
           'No Outbound Queue found in the system'      LENGTH 40
           ##NO_TEXT.

DATA: ls_input  TYPE tab512,
      ls_output TYPE tab512,
      lv_action TYPE char20,
      lv_qcount TYPE char4,
      lt_qview  TYPE TABLE OF trfcqview,
      ls_qview  TYPE trfcqview.

LOAD-OF-PROGRAM.

FORM get_config
              TABLES lt_input  STRUCTURE tab512
                     lt_output STRUCTURE tab512.

  READ TABLE lt_input INTO ls_input INDEX 1.
  CHECK sy-subrc EQ 0.
  lv_action = ls_input.      " Action

  CASE lv_action.

    WHEN lc_export.               " When action is EXPORT

      " FM to get Outbound Queues in SMQ2
      CALL FUNCTION 'TRFC_QOUT_GET_CURRENT_QUEUES'
        EXPORTING
          qname  = '*'
          dest   = '*'
          client = '*'
        TABLES
          qview  = lt_qview.

       " Adding title to the output
        ls_output = lc_title.
        APPEND ls_output TO lt_output.

      IF lt_qview IS NOT INITIAL.

        " Addding heading to the output
        CONCATENATE lc_client      lc_sep
                    lc_queue_name  lc_sep
                    lc_destination lc_sep
                    lc_entries
                    INTO ls_output.
        APPEND ls_output TO lt_output.

        LOOP AT lt_qview INTO ls_qview.

          lv_qcount = ls_qview-qdeep.

          CONDENSE lv_qcount.

          CONCATENATE ls_qview-mandt  lc_sep
                      ls_qview-qname  lc_sep
                      ls_qview-dest   lc_sep
                      lv_qcount
                      INTO ls_output.
          APPEND ls_output TO lt_output.

        ENDLOOP.
        ELSE.
         " No Outbound Queue found in the system
          ls_output = lc_err_msg.
          APPEND ls_output TO lt_output.

      ENDIF.
  ENDCASE.
ENDFORM.
