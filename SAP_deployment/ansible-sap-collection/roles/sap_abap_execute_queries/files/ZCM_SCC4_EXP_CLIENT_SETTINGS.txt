REPORT zcm_scc4_exp_client_settings.

CONSTANTS: lc_sep        TYPE c VALUE '|'               LENGTH 1,
           lc_null       TYPE c VALUE 'NULL'            LENGTH 4,
           lc_export     TYPE c VALUE 'EXPORT'          LENGTH 6,
           lc_mandt      TYPE c VALUE 'mandt'           LENGTH 6,
           lc_mtext      TYPE c VALUE 'mtext'           LENGTH 6,
           lc_ort01      TYPE c VALUE 'ort01'           LENGTH 6,
           lc_mwaer      TYPE c VALUE 'mwaer'           LENGTH 10,
           lc_adrnr      TYPE c VALUE 'adrnr'           LENGTH 10,
           lc_cccategory TYPE c VALUE 'cccategory'      LENGTH 15,
           lc_cccoractiv TYPE c VALUE 'cccoractiv'      LENGTH 15,
           lc_ccnocliind TYPE c VALUE 'ccnocliind'      LENGTH 15,
           lc_cccopylock TYPE c VALUE 'cccopylock'      LENGTH 15,
           lc_ccnocascad TYPE c VALUE 'ccnocascad'      LENGTH 15,
           lc_ccsoftlock TYPE c VALUE 'ccsoftlock'      LENGTH 15,
           lc_ccorigcont TYPE c VALUE 'ccorigcont'      LENGTH 15,
           lc_ccimaildis TYPE c VALUE 'ccimaildis'      LENGTH 15,
           lc_cctemplock TYPE c VALUE 'cctemplock'      LENGTH 15,
           lc_changeuser TYPE c VALUE 'changeuser'      LENGTH 15,
           lc_changedate TYPE c VALUE 'changedate'      LENGTH 15,
           lc_logsys     TYPE c VALUE 'logsys'          LENGTH 6,
           lc_title      TYPE c VALUE 'Client Settings' LENGTH 25,
           lc_err_msg    TYPE c VALUE
           'No Client data found in the system'         LENGTH 40
           ##NO_TEXT.

DATA: ls_output TYPE tab512,
      ls_input  TYPE tab512,
      lv_action TYPE char20,
      lt_t000   TYPE TABLE OF t000,
      ls_t000   TYPE t000.

LOAD-OF-PROGRAM.

FORM get_config
              TABLES lt_input  STRUCTURE tab512
                     lt_output STRUCTURE tab512.

  READ TABLE lt_input INTO ls_input INDEX 1.
  CHECK sy-subrc EQ 0.
  lv_action = ls_input.    " Action

  CASE lv_action.

    WHEN lc_export.      " If action is Export

      SELECT * FROM t000
               INTO TABLE lt_t000.

        " Adding title to the output
        ls_output = lc_title.
        APPEND ls_output TO lt_output.

      IF lt_t000 IS NOT INITIAL.

        " Adding heading to the output

        CONCATENATE   lc_mandt      lc_sep
                      lc_mtext      lc_sep
                      lc_ort01      lc_sep
                      lc_mwaer      lc_sep
                      lc_adrnr      lc_sep
                      lc_cccategory lc_sep
                      lc_cccoractiv lc_sep
                      lc_ccnocliind lc_sep
                      lc_cccopylock lc_sep
                      lc_ccnocascad lc_sep
                      lc_ccsoftlock lc_sep
                      lc_ccorigcont lc_sep
                      lc_ccimaildis lc_sep
                      lc_cctemplock lc_sep
                      lc_changeuser lc_sep
                      lc_changedate lc_sep
                      lc_logsys
                      INTO ls_output.
        APPEND ls_output TO lt_output.

        LOOP AT lt_t000 INTO ls_t000.

          IF ls_t000-adrnr IS INITIAL.
            ls_t000-adrnr = lc_null.
          ENDIF.

          IF ls_t000-cccoractiv IS INITIAL.
            ls_t000-cccoractiv = lc_null.
          ENDIF.

          IF ls_t000-ccnocliind IS INITIAL.
            ls_t000-ccnocliind = lc_null.
          ENDIF.

          IF ls_t000-cccopylock IS INITIAL.
            ls_t000-cccopylock = lc_null.
          ENDIF.

          IF ls_t000-ccnocascad IS INITIAL.
            ls_t000-ccnocascad = lc_null.
          ENDIF.

          IF ls_t000-ccsoftlock IS INITIAL.
            ls_t000-ccsoftlock = lc_null.
          ENDIF.

          IF ls_t000-ccorigcont IS INITIAL.
            ls_t000-ccorigcont = lc_null.
          ENDIF.

          IF ls_t000-cctemplock IS INITIAL.
            ls_t000-cctemplock = lc_null.
          ENDIF.

          IF ls_t000-ccimaildis IS INITIAL.
            ls_t000-ccimaildis = lc_null.
          ENDIF.

          CONCATENATE ls_t000-mandt      lc_sep
                      ls_t000-mtext      lc_sep
                      ls_t000-ort01      lc_sep
                      ls_t000-mwaer      lc_sep
                      ls_t000-adrnr      lc_sep
                      ls_t000-cccategory lc_sep
                      ls_t000-cccoractiv lc_sep
                      ls_t000-ccnocliind lc_sep
                      ls_t000-cccopylock lc_sep
                      ls_t000-ccnocascad lc_sep
                      ls_t000-ccsoftlock lc_sep
                      ls_t000-ccorigcont lc_sep
                      ls_t000-ccimaildis lc_sep
                      ls_t000-cctemplock lc_sep
                      ls_t000-changeuser lc_sep
                      ls_t000-changedate lc_sep
                      ls_t000-logsys
                      INTO ls_output.
          APPEND ls_output TO lt_output.
        ENDLOOP.

      ELSE.
        " No Client data found in the system
        ls_output = lc_err_msg.
        APPEND ls_output TO lt_output.

      ENDIF.

  ENDCASE.

ENDFORM.
